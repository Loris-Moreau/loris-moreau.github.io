Part 9: Positionable Camera

Field of View

Field of view (FOV) is the angle the camera sees from top to bottom. We express it in degrees and derive the viewport height from it :

Camera.h

class Camera
{
public:
    double vFov         = 90;    // Vertical field-of-view in degrees
    Position lookFrom   = Position(0, 0, 0);  // Camera position
    Position lookAt     = Position(0, 0, -1); // Target point
    Vector3  vUp        = Vector3(0, 1, 0);   // Camera-relative "up"

    double defocusAngle = 0;  // Variation angle of rays through each pixel
    double focusDistance = 10; // Distance from camera to the focus plane

    int width           = 100;
    int sampleCount     = 10;
    int maxBounceDepth  = 10;
    double aspectRatio  = 1.0;

    void Render(const Hittable& rWorld);

private:
    int    height;
    double theta, h;
    double viewportHeight, viewportWidth;
    double defocusDiskU, defocusDiskV;

    Position center, originPixelLocation;
    Vector3  pixelDeltaX, pixelDeltaY;
    Vector3  u, v, w; // camera basis vectors

    void     Initialize();
    Color    RayColor(const Ray& rRay, int depth, const Hittable& rWorld) const;
    Ray      GetRay(int x, int y) const;
    Vector3  PixelSampleSquared() const;
    Position DefocusDiskSample() const;
};

Camera Orientation

We build an orthonormal basis from the look direction and the world-up vector. w points toward the camera (opposite the look direction), u points right, and v points up in camera space :

Camera.cpp: Initialize()

void Camera::Initialize()
{
    height = static_cast<int>(width / aspectRatio);
    if (height < 1) height = 1;
    center = lookFrom;

    // Derive viewport dimensions from vFov
    theta          = DegToRad(vFov);
    h              = tan(theta / 2);
    viewportHeight = 2 * h * focusDistance;
    viewportWidth  = viewportHeight * (static_cast<double>(width) / height);

    // Camera basis vectors
    w = Unit(lookFrom - lookAt);
    u = Unit(Cross(vUp, w));
    v = Cross(w, u);

    // Viewport edge vectors
    Vector3 viewportU = viewportWidth  * u;
    Vector3 viewportV = viewportHeight * -v;

    pixelDeltaX = viewportU / width;
    pixelDeltaY = viewportV / height;

    Vector3 viewportOrigin = center
                           - (focusDistance * w)
                           - viewportU / 2
                           - viewportV / 2;
    originPixelLocation = viewportOrigin + 0.5 * (pixelDeltaX + pixelDeltaY);

    // Defocus disk basis vectors
    double defocusRadius = focusDistance * tan(DegToRad(defocusAngle / 2));
    defocusDiskU = u * defocusRadius;
    defocusDiskV = v * defocusRadius;
}

Defocus Blur (Depth of Field)

Real cameras focus on a plane at a given distance. Objects in front of or behind that plane are blurred. We simulate this by scattering each ray's origin within a small disk centered on the camera, the defocus disk :

Vector3.h: random in unit disk

inline Vector3 RandomInUnitDisk()
{
    while (true)
    {
        Vector3 p = Vector3(RandomDouble(-1, 1), RandomDouble(-1, 1), 0);
        if (p.SquaredLength() < 1) return p;
    }
}

Camera.cpp

Ray Camera::GetRay(int x, int y) const
{
    Vector3 pixelCenter = originPixelLocation + (x * pixelDeltaX) + (y * pixelDeltaY);
    Vector3 pixelSample = pixelCenter + PixelSampleSquared();

    // Scatter ray origin across the defocus disk for depth-of-field
    Position rayOrigin = (defocusAngle <= 0) ? center : DefocusDiskSample();
    Vector3  rayDirection = pixelSample - rayOrigin;

    return Ray(rayOrigin, rayDirection);
}

Position Camera::DefocusDiskSample() const
{
    Vector3 p = RandomInUnitDisk();
    return center + (p[0] * defocusDiskU) + (p[1] * defocusDiskV);
}

Final Scene: Book 1

The final scene places hundreds of randomly colored spheres on a ground plane, with three feature spheres in the center. Set the camera to look across the scene with a shallow depth of field :

Raytracing.cpp

int main()
{
    HittableCollection world;

    auto matGround = make_shared<Lambertian>(Color(0.5, 0.5, 0.5));
    world.Add(make_shared<Sphere>(Position(0,-1000,0), 1000, matGround));

    // Random small spheres
    for (int a = -11; a < 11; a++)
    {
        for (int b = -11; b < 11; b++)
        {
            double chooseMat = RandomDouble();
            Position center(a + 0.9*RandomDouble(), 0.2, b + 0.9*RandomDouble());

            if ((center - Position(4,0.2,0)).Length() > 0.9)
            {
                shared_ptr<Material> sphereMat;
                if (chooseMat < 0.8) {
                    // Diffuse
                    auto albedo = Color::Random() * Color::Random();
                    sphereMat = make_shared<Lambertian>(albedo);
                } else if (chooseMat < 0.95) {
                    // Metal
                    auto albedo = Color::Random(0.5, 1);
                    auto fuzz   = RandomDouble(0, 0.5);
                    sphereMat = make_shared<Metal>(albedo, fuzz);
                } else {
                    // Glass
                    sphereMat = make_shared<Dielectric>(1.5);
                }
                world.Add(make_shared<Sphere>(center, 0.2, sphereMat));
            }
        }
    }

    // Feature spheres
    world.Add(make_shared<Sphere>(Position( 0, 1, 0), 1.0, make_shared<Dielectric>(1.5)));
    world.Add(make_shared<Sphere>(Position(-4, 1, 0), 1.0, make_shared<Lambertian>(Color(0.4, 0.2, 0.1))));
    world.Add(make_shared<Sphere>(Position( 4, 1, 0), 1.0, make_shared<Metal>(Color(0.7, 0.6, 0.5), 0.0)));

    Camera camera;
    camera.aspectRatio   = 16.0 / 9.0;
    camera.width         = 1200;
    camera.sampleCount   = 500;
    camera.maxBounceDepth= 50;

    camera.vFov          = 20;
    camera.lookFrom      = Position(13, 2, 3);
    camera.lookAt        = Position( 0, 0, 0);
    camera.vUp           = Vector3(0, 1, 0);

    camera.defocusAngle  = 0.6;
    camera.focusDistance = 10.0;

    camera.Render(world);
    return 0;
}
